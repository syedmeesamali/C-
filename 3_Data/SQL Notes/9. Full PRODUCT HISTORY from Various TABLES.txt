SELECT sin.Prod_ID, sin.Input - sout.SOLD AS [Totals]
FROM 
(SELECT Prod_ID, SUM(Units) AS Input
FROM StockinTable
GROUP BY Prod_ID) sin
LEFT JOIN
    (SELECT Prod_ID, SUM(Boxes) AS SOLD
      FROM StockoutTable
      GROUP BY Prod_ID
     ) sout
     ON sout.Prod_ID = sin.Prod_ID
WHERE sin.Prod_ID = 'A15'
